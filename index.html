Q.1 :Count Characters You are given a string S , and your task is to return an array B( having a size of 2 ) , where B [ 0 ] contains the count of character A (uppercase) in string S and B [ 1 ] contains the count of character D (uppercase) in string S.
Note: You have to complete countCharacters function. No need to take any input.
                    
            
const countCharacters = (S) => 
{
  
 var B = [];
 
  B[0] = [...S].filter(ch => ch == 'A').length;
  B[1] = [...S].filter(ch => ch == 'D').length;
  
  return B;
   
};
Output:- 
        Sample Input
        AbaDd

Note: You have to complete Plain_Check function. No need to take any input. 
              
            
 
     
var Palin_Check = (str) =>
{
	 let a = 0;
	 let b = str.length - 1;
	 
	 while(a < b)
	  {
	   if(str[a] === str[b])
	     {
	       a++;
	       b--;
	     }
	     else 
	       {
	         return "False";
	       }

	  return "True";
}
